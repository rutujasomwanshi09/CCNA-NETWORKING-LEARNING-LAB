########## SWITCH 1 CONFIG ##########

enable
conf t

vlan 10
name IT
exit

vlan 20
name HR
exit

vlan 99
name NATIVE
exit

! Assign access port
interface fa0/3
switchport mode access
switchport access vlan 10
exit

! Configure trunk ports
interface range fa0/1-2
switchport mode trunk
switchport trunk native vlan 99
exit

do wr
